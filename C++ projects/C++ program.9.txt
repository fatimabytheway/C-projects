//program.9:Write a program for linear search.//
#include <iostream>
using namespace std;

class Rational {
private:
    int num, den;
    
    int gcd(int a, int b) {
        while (b != 0) {
            int temp = b;
            b = a % b;
            a = temp;
        }
        return a;
    }
    
    void simplify() {
        if (den < 0) {
            num = -num;
            den = -den;
        }
        int g = gcd(abs(num), den);
        num /= g;
        den /= g;
    }
    
public:
    Rational(int n = 0, int d = 1) {
        if (d == 0) {
            cout << "Denominator cannot be zero!" << endl;
            num = 0;
            den = 1;
            return;
        }
        num = n;
        den = d;
        simplify();
    }
    
    Rational operator+(const Rational& r) {
        return Rational(num * r.den + r.num * den, den * r.den);
    }
    
    friend ostream& operator<<(ostream& os, const Rational& r) {
        if (r.num == 0) os << "0";
        else if (r.den == 1) os << r.num;
        else os << r.num << "/" << r.den;
        return os;
    }
};

int main() {
    Rational r1(1, 2);
    Rational r2(1, 3);
    Rational r3(3, 4);
    
    cout << "r1 = " << r1 << endl;
    cout << "r2 = " << r2 << endl;
    cout << "r3 = " << r3 << endl;
    
    cout << "r1 + r2 = " << (r1 + r2) << endl;
    cout << "r1 + r3 = " << (r1 + r3) << endl;
    cout << "r2 + r3 = " << (r2 + r3) << endl;
    
    return 0;
}
